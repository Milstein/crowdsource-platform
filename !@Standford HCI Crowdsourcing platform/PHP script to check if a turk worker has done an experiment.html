function check_mturk_id($tables, $id) {

	for($i=0; $i < count($tables); $i++){

		$result = mysql_query('SELECT DISTINCT '.mysql_real_escape_string($col_id).' FROM '.mysql_real_escape_string($tables[$i]));

		while($row = mysql_fetch_array($result))

		{

			if($row["mturk_id"] == $id)

			{

				return false;

			}

		}

	}

	

	return true;

}
